<ReleaseNoteSection title="No title">
    <ReleaseNoteDetail type="added">
        * New "transaction_amount_percentage" field added to the split endpoint. This field describes the percentage of the transaction amount that should be split.
    </ReleaseNoteDetail>

    <ReleaseNoteDetail type="added">
        * New response fields added to List Settled Transactions endpoint:
            * order.reference
            * deposit_amount
            * payment_method.card.funding
    </ReleaseNoteDetail>

    <ReleaseNoteDetail type="added">
        * New query parameters added to List Settled Transactions endpoint:
            * order.reference
            * payment_method.card.funding
    </ReleaseNoteDetail>

    <ReleaseNoteDetail type="added">
        * Enhanced the Refund a Sale endpoint to support creating a refund for an Open Banking transaction
    </ReleaseNoteDetail>
</ReleaseNoteSection>

<ReleaseNoteSection title="Disputes">
    <ReleaseNoteDetail type="added">
        * New response "order.reference" field added to List Disputes endpoint
    </ReleaseNoteDetail>

    <ReleaseNoteDetail type="added">
        * New query parameter "order.reference" added to List Disputes endpoint
    </ReleaseNoteDetail>
</ReleaseNoteSection>
